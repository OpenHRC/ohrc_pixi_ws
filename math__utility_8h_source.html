26// check a point in a convex area
27inline bool inConvex(Eigen::Vector2d p, std::vector<Eigen::Vector2d> apexes) {
28 // extend 3d (add z componet for cross product below)
29 Eigen::Vector3d p_3d = Eigen::Vector3d::Zero();
30 p_3d.head(2) = p;
31
32 std::vector<Eigen::Vector3d> apexes_3d(apexes.size(), Eigen::Vector3d::Zero());
33 for (unsigned int i = 0; i < apexes_3d.size(); i++)
34 apexes_3d[i].head(2) = apexes[i];
35
36 int count = 0;
37 apexes_3d.push_back(apexes_3d[0]); // copy first vec at last
38 for (unsigned int i = 0; i < apexes_3d.size() - 1; i++) {
39 if ((p_3d - apexes_3d[i]).cross(apexes_3d[i + 1] - apexes_3d[i])(2) > 0)
40 count++;
41 }
42
43 if (count == apexes_3d.size() - 1 || count == 0) // all cross products are same sign (all positive or all negative)
44 return true;
45 else
46 return false;
47}

152 inline Eigen::VectorXd RungeKuttaIntegrate(const double& t, const Eigen::VectorXd& x, const std::unordered_map<std::string, Eigen::MatrixXd>& u) {
153 Eigen::VectorXd k1 = f(t, x, u);
154 Eigen::VectorXd k2 = f(t + dt * 0.5, x + dt * 0.5 * k1, u);
155 Eigen::VectorXd k3 = f(t + dt * 0.5, x + dt * 0.5 * k2, u);
156 Eigen::VectorXd k4 = f(t + dt, x + dt * k3, u);
157
158 return x + dt / 6.0 * (k1 + 2.0 * k2 + 2.0 * k3 + k4);
159 }
