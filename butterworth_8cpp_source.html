23 double wc = 2.0 / T * tan(2.0 * M_PI * cutoff_freq * T / 2.0); // cutoff prewarp frequency

26 b << (T * T) * (wc * wc) / den, (T * T) * (wc * wc) * 2.0 / den, (T * T) * (wc * wc) / den;
